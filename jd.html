1. import java.sql.*; public class JDBCDemo { public static void main(String[] args) { try { Class.forName("com.mysql.jdbc.Driver"); Connection conn = DriverManager.getConnection("jdbc:mysql://localhost/mydatabase", "root", "password"); Statement stmt = conn.createStatement(); String sql = "SELECT * FROM mytable"; boolean isResultSet = stmt.execute(sql); if (isResultSet) { // If the statement returns a ResultSet, process it ResultSet rs = stmt.getResultSet(); while (rs.next()) { // Retrieve column values from the current row int id = rs.getInt("id"); String name = rs.getString("name"); int age = rs.getInt("age"); System.out.println("ID: " + id + ", Name: " + name + ", Age: " + age); } rs.close(); } else { // If the statement does not return a ResultSet, retrieve the update count int updateCount = stmt.getUpdateCount(); System.out.println("Update count: " + updateCount); } // Close the statement and connection stmt.close(); conn.close(); } catch (SQLException se) { // Handle errors for JDBC se.printStackTrace(); } catch (Exception e) { // Handle errors for Class.forName e.printStackTrace(); } } }

7) import java.sql.*; public class JDBCDemo { public static void main(String[] args) { try { // Register the JDBC driver Class.forName("com.mysql.jdbc.Driver"); // Open a connection Connection conn = DriverManager.getConnection("jdbc:mysql://localhost/mydatabase", "root", "password"); // Create a callable statement to call a stored procedure String sql = "{CALL myproc(?, ?)}"; CallableStatement cstmt = conn.prepareCall(sql); cstmt.setInt(1, 10); // set the value of the first parameter cstmt.registerOutParameter(2, Types.INTEGER); // register the second parameter as an output parameter // Execute the stored procedure cstmt.execute(); // Get the output value of the stored procedure int result = cstmt.getInt(2); System.out.println("Result: " + result); // Close the statement and connection cstmt.close(); conn.close(); } catch (SQLException se) { // Handle errors for JDBC se.printStackTrace(); } catch (Exception e) { // Handle errors for Class.forName e.printStackTrace(); } } }
